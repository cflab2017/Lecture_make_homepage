# 브랜치 전략 단계별 실습 시나리오

## 시나리오: feature/login 브랜치 만들고 병합하기

팀 프로젝트에서 로그인 기능을 구현하는 상황을 시뮬레이션합니다.

---

## 단계 0. 실습 환경 준비

```bash
# 새 디렉토리 생성 및 Git 초기화
mkdir branch-practice
cd branch-practice
git init
git config user.name "실습자"
git config user.email "test@example.com"

# 초기 파일 생성 및 첫 커밋
echo "# 팀 프로젝트" > README.md
git add README.md
git commit -m "init: 프로젝트 초기화"

# main 브랜치 이름 확인
git branch
# 출력: * main  (또는 master — 설정에 따라 다름)
```

---

## 단계 1. 메인 브랜치에 기본 작업 추가

```bash
# index.html 생성
cat > index.html << 'EOF'
<!DOCTYPE html>
<html>
  <head><title>팀 프로젝트</title></head>
  <body>
    <h1>홈 페이지</h1>
  </body>
</html>
EOF

git add index.html
git commit -m "feat: 홈페이지 HTML 추가"

# 현재 상태 확인
git log --oneline
# 출력:
# abc1234 feat: 홈페이지 HTML 추가
# def5678 init: 프로젝트 초기화
```

---

## 단계 2. feature/login 브랜치 생성

```bash
# main에서 feature/login 브랜치 생성 + 이동
git switch -c feature/login

# 브랜치 목록 확인 (* 표시가 현재 브랜치)
git branch
# 출력:
# * feature/login
#   main

# HEAD 확인
git log --oneline --graph --all
# 출력:
# * abc1234 (HEAD -> feature/login, main) feat: 홈페이지 HTML 추가
# * def5678 init: 프로젝트 초기화
```

---

## 단계 3. feature/login 브랜치에서 작업

```bash
# 로그인 페이지 생성
cat > login.html << 'EOF'
<!DOCTYPE html>
<html>
  <head><title>로그인</title></head>
  <body>
    <h1>로그인</h1>
    <form>
      <input type="text" placeholder="아이디" />
      <input type="password" placeholder="비밀번호" />
      <button type="submit">로그인</button>
    </form>
  </body>
</html>
EOF

git add login.html
git commit -m "feat: 로그인 페이지 HTML 추가"

# 로그인 스타일 추가
cat > login.css << 'EOF'
form {
  display: flex;
  flex-direction: column;
  gap: 10px;
  max-width: 300px;
}
button {
  background-color: #3b82f6;
  color: white;
  border: none;
  padding: 10px;
  cursor: pointer;
}
EOF

git add login.css
git commit -m "style: 로그인 페이지 스타일 추가"

# 현재 상태 확인
git log --oneline --graph --all
# 출력:
# * xyz9876 (HEAD -> feature/login) style: 로그인 페이지 스타일 추가
# * uvw4321 feat: 로그인 페이지 HTML 추가
# * abc1234 (main) feat: 홈페이지 HTML 추가
# * def5678 init: 프로젝트 초기화
```

---

## 단계 4. main 브랜치에서 독립적으로 작업 (3-way merge 시뮬레이션)

```bash
# main 브랜치로 이동
git switch main

# main에도 새 커밋 추가 (feature/login과 독립적인 작업)
cat > about.html << 'EOF'
<!DOCTYPE html>
<html>
  <head><title>소개</title></head>
  <body>
    <h1>팀 소개</h1>
    <p>저희는 열심히 개발하는 팀입니다.</p>
  </body>
</html>
EOF

git add about.html
git commit -m "feat: 소개 페이지 추가"

# 분기된 구조 확인
git log --oneline --graph --all
# 출력:
# * 111aaaa (HEAD -> main) feat: 소개 페이지 추가
# | * xyz9876 (feature/login) style: 로그인 페이지 스타일 추가
# | * uvw4321 feat: 로그인 페이지 HTML 추가
# |/
# * abc1234 feat: 홈페이지 HTML 추가
# * def5678 init: 프로젝트 초기화
```

---

## 단계 5. feature/login을 main에 병합 (3-way merge)

```bash
# main에서 feature/login 병합
git merge feature/login
# 3-way merge이므로 병합 커밋이 생성됨
# 에디터가 열리면 커밋 메시지 확인 후 저장

# 병합 결과 확인
git log --oneline --graph --all
# 출력:
# *   222bbbb (HEAD -> main) Merge branch 'feature/login'
# |\
# | * xyz9876 (feature/login) style: 로그인 페이지 스타일 추가
# | * uvw4321 feat: 로그인 페이지 HTML 추가
# * | 111aaaa feat: 소개 페이지 추가
# |/
# * abc1234 feat: 홈페이지 HTML 추가
# * def5678 init: 프로젝트 초기화

# 파일 확인 — 두 브랜치의 파일이 모두 있어야 함
ls
# 출력: README.md  about.html  index.html  login.css  login.html
```

---

## 단계 6. 병합 완료 후 브랜치 삭제

```bash
# 병합된 브랜치 삭제
git branch -d feature/login

# 브랜치 목록 확인 — feature/login이 사라져야 함
git branch
# 출력: * main

# 최종 상태
git log --oneline --graph
```

---

## 실습 완료 체크리스트

- [ ] `git switch -c feature/login`으로 브랜치를 만들고 이동했다.
- [ ] feature/login에서 2개 이상의 커밋을 했다.
- [ ] main으로 돌아와서 독립적인 커밋을 했다.
- [ ] `git merge feature/login`으로 병합했다.
- [ ] `git log --oneline --graph --all`로 브랜치 구조를 확인했다.
- [ ] `git branch -d feature/login`으로 브랜치를 삭제했다.
