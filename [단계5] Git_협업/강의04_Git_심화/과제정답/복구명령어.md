# 강의04 과제 정답 — 커밋 히스토리 정리 및 태그 붙이기

## 전체 과정 및 명령어 설명

### 1단계: 실습 환경 준비

```bash
mkdir git-advanced-practice
cd git-advanced-practice
git init
git config user.name "홍길동"
git config user.email "hong@example.com"

# 초기 커밋
echo "# 쇼핑몰 프로젝트" > README.md
git add README.md
git commit -m "init: 프로젝트 초기화"
# [main (root-commit) aaa0001] init: 프로젝트 초기화

# 지저분한 커밋들 순서대로 만들기
echo "<h1>상품 목록</h1><ul id='product-list'></ul>" > products.html
git add products.html
git commit -m "WIP: 상품 목록 작업 중"
# [main bbb0002] WIP: 상품 목록 작업 중

echo ".title { color: blue; font-size: 24px; }" > style.css
git add style.css
git commit -m "임시 저장"
# [main ccc0003] 임시 저장

echo "const getProducts = () => [];" > products.js
git add products.js
git commit -m "WIP2: 함수 추가"
# [main ddd0004] WIP2: 함수 추가

echo "const filterProducts = (query) => getProducts().filter(p => p.name.includes(query));" >> products.js
git add products.js
git commit -m "완성!"
# [main eee0005] 완성!

# 현재 지저분한 히스토리 확인
git log --oneline
# eee0005 (HEAD -> main) 완성!
# ddd0004 WIP2: 함수 추가
# ccc0003 임시 저장
# bbb0002 WIP: 상품 목록 작업 중
# aaa0001 init: 프로젝트 초기화
```

### 2단계: git reset --soft로 커밋 합치기

```bash
# HEAD~4 = 4개 커밋 전 = init 커밋 위치
# --soft: 커밋은 취소하지만 모든 변경사항을 스테이징 상태로 유지
git reset --soft HEAD~4
# HEAD는 이제 init 커밋(aaa0001)을 가리킴

# 현재 상태 확인
git status
# On branch main
# Changes to be committed:  ← 4개 커밋의 모든 변경사항이 스테이징됨
#   new file: products.html
#   new file: products.js
#   new file: style.css

git log --oneline
# aaa0001 (HEAD -> main) init: 프로젝트 초기화
# (나머지 4개 커밋은 사라짐, 변경사항은 스테이징에 남아있음)

# 4개 커밋을 하나의 의미 있는 커밋으로 만들기
git commit -m "feat: 상품 목록 페이지 구현 (HTML/CSS/JS)

- products.html: 상품 목록 기본 구조 추가
- style.css: 타이틀 스타일 정의
- products.js: getProducts, filterProducts 함수 구현"
# [main fff0006] feat: 상품 목록 페이지 구현 (HTML/CSS/JS)

# 정리된 히스토리 확인
git log --oneline
# fff0006 (HEAD -> main) feat: 상품 목록 페이지 구현 (HTML/CSS/JS)
# aaa0001 init: 프로젝트 초기화
# (깔끔하게 2개만 남음!)
```

### 3단계: git tag로 버전 표시

```bash
# 주석 태그(annotated tag) 생성 — 작성자, 날짜, 메시지 포함 (권장)
git tag -a v1.0.0 -m "상품 목록 기능 첫 번째 릴리즈

포함 기능:
- 상품 목록 HTML 구조
- 기본 스타일
- 상품 조회/필터 함수"

# 태그 목록 확인
git tag
# v1.0.0

# 태그 상세 정보 확인
git show v1.0.0
# tag v1.0.0
# Tagger: 홍길동 <hong@example.com>
# Date:   ...
#
# 상품 목록 기능 첫 번째 릴리즈
# ...
#
# commit fff0006 (HEAD -> main, tag: v1.0.0)
# Author: 홍길동 <hong@example.com>
# ...
# feat: 상품 목록 페이지 구현 (HTML/CSS/JS)

# 최종 히스토리
git log --oneline
# fff0006 (HEAD -> main, tag: v1.0.0) feat: 상품 목록 페이지 구현 (HTML/CSS/JS)
# aaa0001 init: 프로젝트 초기화
```

## 보너스: git stash 실습

```bash
# 새 기능 작업 중에 긴급 수정 요청이 왔다고 가정
echo "const searchProducts = (q) => [];" >> products.js

# 아직 커밋하기엔 미완성 — stash로 임시 저장
git stash push -m "검색 기능 작업 중 (미완성)"
# Saved working directory and index state On main: 검색 기능 작업 중 (미완성)

git status
# nothing to commit  ← 작업 내용이 임시 저장됨

# stash 목록 확인
git stash list
# stash@{0}: On main: 검색 기능 작업 중 (미완성)

# 긴급 수정 시뮬레이션
git switch -c hotfix/typo-fix
echo "/* 오타 수정 */" >> style.css
git add style.css
git commit -m "fix: CSS 주석 오타 수정"

git switch main
git merge hotfix/typo-fix --no-ff -m "merge: 긴급 오타 수정 반영"
git branch -d hotfix/typo-fix

# 다시 원래 작업으로 복귀
git stash pop
# On branch main
# Changes not staged for commit:
#   modified: products.js   ← 임시 저장했던 내용 복원됨

git status
# modified: products.js  ← 작업 계속 가능
```

## 보너스: git cherry-pick 실습

```bash
# 다른 브랜치에서 특정 커밋만 가져오는 상황
git switch -c feature/payment
echo "const processPayment = () => {};" > payment.js
git add payment.js
git commit -m "feat: 결제 처리 함수 추가"

echo "const validateCard = () => {};" >> payment.js
git add payment.js
git commit -m "feat: 카드 유효성 검사 추가"

# 두 커밋의 해시 확인
git log --oneline
# hhh0009 feat: 카드 유효성 검사 추가
# ggg0008 feat: 결제 처리 함수 추가
# fff0006 feat: 상품 목록 페이지 구현

# main으로 돌아와서 두 번째 커밋(validateCard)만 가져오기
git switch main
git cherry-pick hhh0009
# [main iii0010] feat: 카드 유효성 검사 추가  ← 새 해시로 복사됨

git log --oneline
# iii0010 (HEAD -> main) feat: 카드 유효성 검사 추가  ← cherry-pick된 커밋
# fff0006 feat: 상품 목록 페이지 구현
# aaa0001 init: 프로젝트 초기화
```

## 명령어 정리

| 명령어 | 설명 |
|--------|------|
| `git reset --soft HEAD~N` | N개 커밋 취소, 변경사항은 staged 유지 |
| `git stash push -m "메시지"` | 현재 변경사항 임시 저장 |
| `git stash pop` | 마지막 stash 복원 + 목록에서 제거 |
| `git tag -a v1.0.0 -m "메시지"` | 주석 태그 생성 |
| `git show v1.0.0` | 태그 상세 정보 확인 |
| `git cherry-pick <해시>` | 특정 커밋만 현재 브랜치에 적용 |
| `git reflog` | HEAD 이동 이력 — 복구에 사용 |
