# 강의01 과제 정답 — 두 기능 브랜치 동시 개발 실습 결과

## 전체 명령어 순서

### 1단계: 준비

```bash
# 1-1. 새 디렉토리 생성 및 Git 초기화
mkdir branch-assignment
cd branch-assignment
git init
git config user.name "홍길동"
git config user.email "hong@example.com"

# 1-2. 초기 index.html 생성
cat > index.html << 'EOF'
<!DOCTYPE html>
<html lang="ko">
  <head>
    <meta charset="UTF-8">
    <title>브랜치 실습</title>
  </head>
  <body>
    <main>
      <p>메인 콘텐츠</p>
    </main>
  </body>
</html>
EOF

# 1-3. 첫 커밋
git add index.html
git commit -m "init: 프로젝트 초기화"
# 출력: [main (root-commit) a1b2c3d] init: 프로젝트 초기화

# 현재 상태 확인
git log --oneline
# a1b2c3d (HEAD -> main) init: 프로젝트 초기화
```

### 2단계: feature/header 작업

```bash
# 2-1. main에서 feature/header 브랜치 생성 + 이동
git switch -c feature/header
# Switched to a new branch 'feature/header'

# 브랜치 확인
git branch
# * feature/header
#   main

# 2-2. index.html에 헤더 추가
# (기존 <body> 태그 안, <main> 위에 <header> 추가)
cat > index.html << 'EOF'
<!DOCTYPE html>
<html lang="ko">
  <head>
    <meta charset="UTF-8">
    <title>브랜치 실습</title>
  </head>
  <body>
    <header>
      <nav>
        <a href="/">홈</a>
        <a href="/about">소개</a>
        <a href="/contact">연락처</a>
      </nav>
    </header>
    <main>
      <p>메인 콘텐츠</p>
    </main>
  </body>
</html>
EOF

# 2-3. 커밋
git add index.html
git commit -m "feat: 헤더 네비게이션 추가"
# [feature/header d4e5f6g] feat: 헤더 네비게이션 추가

# 상태 확인
git log --oneline --graph --all
# * d4e5f6g (HEAD -> feature/header) feat: 헤더 네비게이션 추가
# * a1b2c3d (main) init: 프로젝트 초기화
```

### 3단계: feature/footer 작업

```bash
# 3-1. main으로 복귀 (feature/header의 변경사항 없이 원래 상태로)
git switch main
# Switched to branch 'main'

# index.html 확인 — <header>가 없어야 함 (main에는 아직 없음)
cat index.html
# (header 없는 원래 버전)

# 3-2. feature/footer 브랜치 생성 + 이동
git switch -c feature/footer
# Switched to a new branch 'feature/footer'

# 3-3. index.html에 푸터 추가
cat > index.html << 'EOF'
<!DOCTYPE html>
<html lang="ko">
  <head>
    <meta charset="UTF-8">
    <title>브랜치 실습</title>
  </head>
  <body>
    <main>
      <p>메인 콘텐츠</p>
    </main>
    <footer>
      <p>© 2025 팀 프로젝트. All rights reserved.</p>
      <p>문의: contact@example.com</p>
    </footer>
  </body>
</html>
EOF

# 3-4. 커밋
git add index.html
git commit -m "feat: 푸터 정보 추가"
# [feature/footer h7i8j9k] feat: 푸터 정보 추가

# 분기된 구조 확인
git log --oneline --graph --all
# * h7i8j9k (HEAD -> feature/footer) feat: 푸터 정보 추가
# | * d4e5f6g (feature/header) feat: 헤더 네비게이션 추가
# |/
# * a1b2c3d (main) init: 프로젝트 초기화
```

### 4단계: 순서대로 병합

```bash
# 4-1. main으로 이동
git switch main
# Switched to branch 'main'

# 4-2. feature/header 먼저 병합
git merge feature/header
# Updating a1b2c3d..d4e5f6g
# Fast-forward    ← main에 추가 커밋이 없으므로 Fast-forward 발생
#  index.html | 6 ++++++
#  1 file changed, 6 insertions(+)

git log --oneline --graph --all
# * d4e5f6g (HEAD -> main, feature/header) feat: 헤더 네비게이션 추가
# | * h7i8j9k (feature/footer) feat: 푸터 정보 추가
# |/
# * a1b2c3d init: 프로젝트 초기화

# 4-3. feature/footer 병합
git merge feature/footer
# ⚠️ 충돌 발생!
# Auto-merging index.html
# CONFLICT (content): Merge conflict in index.html
# Automatic merge failed; fix conflicts and then commit the result.

# ── 충돌 해결 ──────────────────────────────────────────────
# index.html을 열면 아래 형태로 충돌 마커가 보임:
#
# <body>
# <<<<<<< HEAD
#     <header>
#       <nav>...</nav>
#     </header>
# =======
# >>>>>>> feature/footer
#     <main>
#       <p>메인 콘텐츠</p>
#     </main>
# <<<<<<< HEAD
# =======
#     <footer>
#       <p>© 2025 ...</p>
#     </footer>
# >>>>>>> feature/footer

# 두 브랜치의 내용을 모두 포함하도록 수동으로 수정:
cat > index.html << 'EOF'
<!DOCTYPE html>
<html lang="ko">
  <head>
    <meta charset="UTF-8">
    <title>브랜치 실습</title>
  </head>
  <body>
    <header>
      <nav>
        <a href="/">홈</a>
        <a href="/about">소개</a>
        <a href="/contact">연락처</a>
      </nav>
    </header>
    <main>
      <p>메인 콘텐츠</p>
    </main>
    <footer>
      <p>© 2025 팀 프로젝트. All rights reserved.</p>
      <p>문의: contact@example.com</p>
    </footer>
  </body>
</html>
EOF

# 충돌 해결 후 스테이징 및 커밋
git add index.html
git commit -m "merge: feature/footer 병합 (header + footer 통합)"
# [main l4m5n6o] merge: feature/footer 병합

# 4-4. 브랜치 삭제
git branch -d feature/header
# Deleted branch feature/header (was d4e5f6g).

git branch -d feature/footer
# Deleted branch feature/footer (was h7i8j9k).

# 최종 상태 확인
git log --oneline --graph
# *   l4m5n6o (HEAD -> main) merge: feature/footer 병합 (header + footer 통합)
# |\
# | * h7i8j9k feat: 푸터 정보 추가
# * | d4e5f6g feat: 헤더 네비게이션 추가
# |/
# * a1b2c3d init: 프로젝트 초기화

git branch
# * main   (feature 브랜치들이 모두 삭제됨)
```

## 결과 요약

| 단계 | 발생한 일 | 설명 |
|------|----------|------|
| feature/header → main | **Fast-forward merge** | main에 추가 커밋 없음 |
| feature/footer → main | **3-way merge + 충돌** | 두 브랜치 모두 index.html 수정 |
| 충돌 해결 | 수동 편집 후 커밋 | header와 footer 내용 모두 포함 |

## 핵심 배운 점

1. 두 브랜치가 **같은 파일의 같은 부분**을 수정하면 충돌이 발생합니다.
2. 충돌은 오류가 아니라 **정상적인 현상**입니다 — 수동으로 해결하면 됩니다.
3. `git log --oneline --graph --all`로 브랜치 구조를 항상 파악하세요.
